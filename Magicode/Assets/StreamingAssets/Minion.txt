    	}

		yield return new WaitForEndOfFrame();
		while(true) {
if(true ) {}if(hasCollided ) {GetComponent<Spell>().ActivateEffect("move_forward");
}	
			yield return new WaitForEndOfFrame();

			}
		}
	}
